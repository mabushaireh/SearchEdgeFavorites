# SearchEdgeFavorites AI Configuration
# =====================================
# Configure AI providers for automatic URL summarization

# OpenAI Configuration
# Get your API key from: https://platform.openai.com/api-keys
OPENAI_API_KEY=your_openai_api_key_here

# Google Gemini Configuration
# Get your API key from: https://aistudio.google.com/app/apikey
GEMINI_API_KEY=your_gemini_api_key_here

# AI Provider Selection
# Options: openai, gemini, both
# - openai: Use only OpenAI GPT-3.5 (default)
# - gemini: Use only Google Gemini Pro
# - both: Try both providers (first successful response wins)
AI_PROVIDER=openai
